```dataviewjs
const targetTag = "#산업안전기사/암기"; 

const pages = dv.pages("#산업안전기사"); 

for (const page of pages) {
    const file = app.vault.getAbstractFileByPath(page.file.path);
    const content = await app.vault.read(file);
    const sections = content.split('---');
    const matches = sections.filter(s => s.includes(targetTag));
    
    if (matches.length > 0) {
        dv.header(2, page.file.name);
        matches.forEach(m => {
            dv.paragraph(m.trim());
            dv.el("hr", ""); 
        });
    }
}
```
